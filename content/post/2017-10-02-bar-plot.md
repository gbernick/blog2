library(sqldf)
library(ggplot2)
library(Lahman)


#Extracting Data

query<-"Select name, HR
from Teams 
where yearID=1980
order by HR"

result<-sqldf(query)

result$name<-factor(result$name, levels=result$name)



#Visualizing Data
  

ggplot()+
  geom_bar(data=result, aes(x=name,y=HR), stat = 'identity', color='blue', fill='white')+
  coord_flip()+
  ylab("Homeruns")+
  xlab("Team")+
  ggtitle("1980 Homerun Totals")
